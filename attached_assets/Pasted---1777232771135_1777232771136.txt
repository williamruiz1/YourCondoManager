  SELECT id, email, role, created_at                                                                                              
  FROM admin_users                    
  WHERE email = 'yourcondomanagement@gmail.com';                                                                                                                                                      
  
  -- Q3: your scope rows                                                                                                                                                                              
  SELECT                                                                                                                          
    s.admin_user_id,                                                                                                                                                                                  
    s.association_id,
    a.name AS association_name,                                                                                                                                                                       
    a.archived_at,                                                                                                                
    s.role AS scope_role,             
    s.created_at AS scope_created_at                                                                                                                                                                  
  FROM admin_association_scopes s
  JOIN admin_users u ON u.id = s.admin_user_id                                                                                                                                                        
  LEFT JOIN associations a ON a.id = s.association_id                                                                                                                                                 
  WHERE u.email = 'yourcondomanagement@gmail.com'
  ORDER BY a.name NULLS LAST;
