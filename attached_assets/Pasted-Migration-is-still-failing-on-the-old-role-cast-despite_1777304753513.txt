  Theory 2: Replit's deployment is caching the old migration plan from before your UPDATE, so it's re-running a plan generated when the bad rows still existed.                                       
                                                                                                                                                                                                      
  Check theory 1 (definitive — uses pg_attribute, more reliable than information_schema)                                                                                                              
                                                                                                                                                                                                      
  psql "$DATABASE_URL" <<'SQL'                                                                                                                                                                        
  -- Every column anywhere using admin_user_role                                                                               
  SELECT n.nspname AS schema, c.relname AS table, a.attname AS column                                                                                                                                 
  FROM pg_attribute a
  JOIN pg_class c ON c.oid = a.attrelid                                                                                                                                                               
  JOIN pg_namespace n ON n.oid = c.relnamespace                                                                                                                                                       
  JOIN pg_type t ON t.oid = a.atttypid
  WHERE t.typname = 'admin_user_role'                                                                                                                                                                 
    AND a.attnum > 0 AND NOT a.attisdropped                                                                                    
    AND c.relkind = 'r'                                                                                                                                                                               
  ORDER BY 1,2,3;                                                                                                              
                                                                                                                                                                                                      
  -- Confirm zero board-admin rows in the known-cleaned places                                                                                                                                        
  SELECT 'admin_users.role' AS where_, count(*) FROM admin_users WHERE role::text = 'board-admin'                                                                                                     
  UNION ALL SELECT 'pcl.old_role', count(*) FROM permission_change_logs WHERE old_role::text = 'board-admin'                                                                                          
  UNION ALL SELECT 'pcl.new_role', count(*) FROM permission_change_logs WHERE new_role::text = 'board-admin';
